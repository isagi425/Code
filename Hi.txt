// @name         OP Cookie Editor (Safe Personal Use)
// @namespace    https://example.com/

// @description  Personal cookie manager with OP UI (view, copy, clear). 100% safe, no stealing.
// @author       You
// @match        *://*/*
// @grant        GM_addStyle
// ==/UserScript==

(function() {
    'use strict';

    // Add Styles
    GM_addStyle(`
      #cookieToggle {
        position: fixed;
        bottom: 20px;
        right: 20px;
        z-index: 99999;
        background: #111;
        border: 2px solid white;
        color: white;
        font-size: 18px;
        padding: 10px 14px;
        border-radius: 50%;
        cursor: pointer;
        font-family: 'Arial Black', sans-serif;
        box-shadow: 0 0 12px rgba(255,255,255,0.7);
        transition: transform 0.2s ease;
      }
      #cookieToggle:hover {
        transform: scale(1.1);
      }
      #cookieEditor {
        position: fixed;
        bottom: 80px;
        right: 20px;
        width: 90%;
        max-width: 400px;
        height: 60%;
        background: #fff;
        border: 3px solid black;
        border-radius: 12px;
        z-index: 99999;
        display: none;
        flex-direction: column;
        box-shadow: 0 0 20px rgba(0,0,0,0.8);
        animation: fadeIn 0.3s ease-out;
      }
      #cookieEditorHeader {
        background: black;
        color: white;
        padding: 10px;
        font-size: 16px;
        text-align: center;
        font-family: 'Arial Black', sans-serif;
        border-bottom: 2px solid white;
        display: flex;
        justify-content: space-between;
        align-items: center;
      }
      #cookieEditorBody {
        flex: 1;
        overflow-y: auto;
        padding: 10px;
        font-family: monospace;
        font-size: 14px;
        white-space: pre-wrap;
      }
      #cookieEditorFooter {
        padding: 8px;
        text-align: center;
        border-top: 2px solid black;
        background: #f2f2f2;
      }
      #cookieEditorFooter button {
        margin: 5px;
        padding: 8px 12px;
        border: none;
        border-radius: 8px;
        cursor: pointer;
        font-weight: bold;
      }
      #copyCookies {
        background: #222;
        color: white;
      }
      #clearCookies {
        background: crimson;
        color: white;
      }
      .closeBtn {
        cursor: pointer;
        font-size: 18px;
        font-weight: bold;
        padding: 0 8px;
      }
      @keyframes fadeIn {
        from {opacity: 0; transform: translateY(20px);}
        to {opacity: 1; transform: translateY(0);}
      }
    `);

    // Add toggle button
    const toggleBtn = document.createElement("div");
    toggleBtn.id = "cookieToggle";
    toggleBtn.innerText = "‚öôÔ∏è";
    document.body.appendChild(toggleBtn);

    // Add editor UI
    const editor = document.createElement("div");
    editor.id = "cookieEditor";
    editor.innerHTML = `
      <div id="cookieEditorHeader">
        <span>üç™ Cookie Editor (Personal Use Only)</span>
        <span class="closeBtn" id="closeCookieEditor">‚úñ</span>
      </div>
      <div id="cookieEditorBody">Loading cookies...</div>
      <div id="cookieEditorFooter">
        <button id="copyCookies">Copy</button>
        <button id="clearCookies">Clear</button>
      </div>
    `;
    document.body.appendChild(editor);

    // Functions
    function refreshCookies() {
      document.getElementById("cookieEditorBody").innerText = document.cookie || "No cookies found.";
    }

    // Toggle open/close
    toggleBtn.addEventListener("click", () => {
      if (editor.style.display === "flex") {
        editor.style.display = "none";
      } else {
        refreshCookies();
        editor.style.display = "flex";
      }
    });

    // Close button
    document.getElementById("closeCookieEditor").addEventListener("click", () => {
      editor.style.display = "none";
    });

    // Copy button
    document.getElementById("copyCookies").addEventListener("click", () => {
      navigator.clipboard.writeText(document.cookie).then(() => {
        alert("Cookies copied!");
      });
    });

    // Clear button
    document.getElementById("clearCookies").addEventListener("click", () => {
      if (confirm("Clear all cookies for this site?")) {
        const cookies = document.cookie.split(";");
        for (let c of cookies) {
          document.cookie = c.split("=")[0].trim() + "=;expires=Thu, 01 Jan 1970 00:00:00 GMT;path=/";
        }
        refreshCookies();
      }
    });
})();
